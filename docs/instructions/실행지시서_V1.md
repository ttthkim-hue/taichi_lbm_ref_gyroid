## OpenLB → Taichi 이식 확정 파라미터

```python
# 모든 케이스 공통 (OpenLB UnitConverter에서 추출·검증된 값)
TAU     = 0.595          # relaxation time (OpenLB 하드코딩값)
NU_PHYS = 3.52e-5        # [m²/s] 200°C 공기 (SCR 운전온도)
RHO     = 0.746          # [kg/m³] 200°C 공기
U_INLET = 0.2778         # [m/s] GHSV=10,000 기준
DX      = 0.0002         # [m] 격자 간격

# 위 값에서 자동 계산
U_LATT  = 0.05           # lattice velocity (OpenLB 고정값)
DT      = U_LATT * DX / U_INLET  # = 3.60e-5 s
NU_LATT = (TAU - 0.5) / 3        # = 0.03167
# 검증: NU_LATT * DX² / DT = 3.517e-5 ≈ NU_PHYS ✅

# 수렴 판정
CHECK_INTERVAL = 500
CONV_EPS = 1e-3
MAX_STEPS = 60000  # transit × 5 이상 확보
```

### 수정 포인트 3가지

**수정 1: 온도 → 200°C**
```python
# 변경 전
nu_phys = 1.516e-5   # 20°C
rho_ref = 1.204

# 변경 후
nu_phys = 3.52e-5    # 200°C SCR 운전온도
rho_ref = 0.746
```

**수정 2: τ 고정 (클램프 제거)**
```python
# 변경 전
tau_eff = max(0.55, tau_computed)  # 자동 클램프

# 변경 후
tau_eff = 0.595  # OpenLB 확정값으로 고정
omega = 1.0 / tau_eff  # = 1.6807
```

**수정 3: dt를 u_latt 기반으로 계산**
```python
# 변경 전
dt = 5e-6  # 임의값

# 변경 후
u_latt = 0.05
dt = u_latt * dx / u_in  # = 0.05 * 0.0002 / 0.2778 = 3.6e-5 s
```

**수정 4: 단위 변환 검증 출력 추가**
```python
# 솔버 시작 시 파라미터 검증 출력
nu_latt = (tau_eff - 0.5) / 3
nu_check = nu_latt * dx**2 / dt
Ma = u_latt / (1/3)**0.5
print(f"[설정 검증]")
print(f"  τ={tau_eff}, ω={1/tau_eff:.4f}")
print(f"  dt={dt:.3e}, u_latt={u_latt:.4f}")
print(f"  Ma={Ma:.4f}  {'✅' if Ma<0.1 else '❌'}")
print(f"  ν_latt={nu_latt:.5f}")
print(f"  ν_check={nu_check:.4e} vs ν_target={nu_phys:.4e}  "
      f"{'✅' if abs(nu_check-nu_phys)/nu_phys<0.01 else '❌'}")
```

**수정 5: 질량보존 모니터링 추가**
```python
# 매 check_interval 스텝마다
flux_in  = rho_in  * u_avg_in  * n_fluid_inlet
flux_out = rho_out * u_avg_out * n_fluid_outlet
mass_imbal = abs(flux_in - flux_out) / flux_in * 100
print(f"  mass_imbalance={mass_imbal:.2f}%  {'✅' if mass_imbal<1 else '❌'}")
```

## STEP 2: 결과 검증

```bash
python3 << 'EOF'
import pandas as pd, math, json

# 로그에서 마지막 구간 읽기
df = pd.read_csv('/mnt/h/taichi_lbm_ref_gyroid/Results_EmptyDuct_200C/simulation_log.csv')
last = df.tail(200).mean()

# 이론값 (200°C, 126×126 단면, L=110mm)
rho=0.746; nu=3.52e-5; mu=nu*rho; u=0.2778
Dh=126*0.0002; L=550*0.0002
Re=rho*u*Dh/mu; f=56.9/Re; dP_th=f*(L/Dh)*0.5*rho*u**2

dP_sim = last.get('dP_pascal', last.get('deltaP', 0))
u_in   = last.get('u_in_sim', last.get('Uin', 0))
u_out  = last.get('u_out_sim', last.get('Uout', 0))
rho_in = last.get('rho_in', 1.0)
rho_out= last.get('rho_out', 1.0)

err_dP  = abs(dP_sim - dP_th)/dP_th*100 if dP_th>0 and dP_sim>0 else 999
mass_imbal = abs(rho_in*u_in - rho_out*u_out)/(rho_in*u_in)*100 if u_in>0 else 999

print(f"=== 빈 덕트 200°C 검증 ===")
print(f"이론 ΔP:      {dP_th:.5f} Pa  (Re={Re:.1f})")
print(f"시뮬 ΔP:      {dP_sim:.5f} Pa  오차={err_dP:.3f}%  {'✅' if err_dP<1 else '❌'}")
print(f"Uin/Uout:     {u_in:.4f}/{u_out:.4f} m/s  (목표={u:.4f}, ±5%)")
print(f"ρ_in/ρ_out:   {rho_in:.4f}/{rho_out:.4f}  (이상적=거의 동일)")
print(f"질량 불균형:  {mass_imbal:.2f}%  {'✅ <1%' if mass_imbal<1 else '❌'}")
print(f"OpenLB ΔP:    0.03595 Pa (오차 0.025%)")
print(f"\n→ {'✅ Level 2 PASS' if err_dP<1 and mass_imbal<1 else '❌ 추가 수정 필요'}")
EOF
```
